  9     /**
 10      * Represents a value that will eventually be set.
 11      * @class
 12      * @private
 13      */
 14     var Promise = function() {
 15         var resolved = false, value;
 16         this.__defineGetter__('resolved', function() {
 17             return resolved;
 18         });
 19         this.__defineGetter__('value', function() {
 20             if ( ! resolved ) {
 21                 throw "Unresolved promise queried";
 22             }
 23             return value;
 24         });
 25         this.__defineSetter__('value', function(val) {
 26             resolved = true;
 27             value = val;
 28         });
 29     };

 31     /**
 32      * A test runner, backed by phantomjs.
 33      *
 34      * @param {String} base_url The base URL that all {@link #open} calls will be relative to.
 35      * @class
 36      */
 37     var Test = function(base_url) {
 38         var running = false;
 39         var page;
 40         var opt = {
 41             base_url: null,
 42             width: 1024,
 43             height: 768,
 44             timeout: 10000
 45         };
 46         var tests = {
 47             total: 0,
 48             ok: 0,
 49             failed: 0
 50         };
 51         var queue = [];
 52 
 53         // initialise
 54         opt.base_url = base_url;
 55 
 56         /** @private */
 57         function proxy(func) {
 58             var self = this;
 59             return function() { return func.apply(self, arguments); };
 60         }
 61 
 62         /** @private */
 63         function ok(success, description, got, expected) {
 64             var n = ++tests.total;
 65             if ( success ) {
 66                 console.log('ok ' + n + ' ' + description);
 67                 tests.ok++;
 68             }
 69             else {
 70                 console.log('not ok ' + n + ' ' + description);
 71                 if ( arguments.length == 4 ) {
 72                     diag("Got: " + got + "\nExpected: " + expected, 1);
 73                 }
 74                 tests.failed++;
 75             }
 76         }
 77 
 78         /** @private */
 79         function diag(message, indent) {
 80             indent = indent || 0;
 81             indent = Array(1+indent).join("\t");
 82             message.split(/\r?\n/).forEach(function(line) {
 83                 console.log('# ' + indent + line);
 84             });
 85         }
 86 
 87         /** @private */
 88         function job_done_factory(source) {
 89             var done = false;
 90             return function() {
 91                 if ( done ) {
 92                     // We will not be done twice!
 93                     return;
 94                 }
 95                 //console.log('job done: ' + source);
 96                 running = false;
 97                 done = true;
 98                 setTimeout(run_queue, 0);
 99             };
100         }
101 
102         /** @private */
103         function run_queue() {
104             if ( running || queue.length === 0 ) {
105                 // Nothing needs doing
106                 return;
107             }
108             running = true;
109             var job = queue.shift();
110             try {
111                 if ( job.is_async ) {
112                     var done = job_done_factory(job.func);
113                     var timeout = setTimeout(function() {
114                         ok(false, job.func + ' timed out');
115                         done();
116                     }, opt.timeout);
117                     job.func.call(this, function() {
118                         clearTimeout(timeout);
119                         done();
120                     });
121                 }
122                 else {
123                     job.func.call(this);
124                     setTimeout(job_done_factory(job.func), 0);
125                 }
126             }
127             catch(e) {
128                 console.log('Internal error: ' + e);
129                 phantom.exit(127);
130             }
131         }
132 
133         /** @private */
134         function queue_async(name, func) {
135             func.toString = function() { return name; };
136             queue.push({
137                 is_async: true,
138                 func: func
139             });
140             run_queue();
141         }
142 
143         /** @private */
144         function queue_sync(name, func) {
145             func.toString = function() { return name; };
146             queue.push({
147                 is_async: false,
148                 func: func
149             });
150             run_queue();
151         }

153         /** @private */
154         function page_eval(arg) {
155             if ( typeof(arg) === 'function' ) {
156                 return page.evaluate(arg);
157             }
158             if ( arg instanceof Promise ) {
159                 return arg.value;
160             }
161             if ( typeof(arg) === 'string' ) {
162                 return arg;
163             }
164             throw "Can't page_eval type: " + typeof(arg);
165         }
166 
167         /** @private */
168         function page_set_argument(arg) {
169             page.evaluate('function() { __testlib_argument = ' + JSON.stringify(arg) + '; }');
170         }
171 
172         /** @private */
173         function invoke_jquery() {
174             var promise = new Promise();
175             var args = Array.prototype.slice.call(arguments);
176             var method = args.shift();
177             var selector = args.shift();
178 
179             queue_sync('invoke_jquery', function() {
180                 page_set_argument({
181                     selector: selector,
182                     method: method,
183                     args: args
184                 });
185                 promise.value = page_eval(function() {
186                     var arg = __testlib_argument;
187                     var obj = $(arg.selector);
188                     var ret = obj[arg.method].apply(obj, arg.args);
189                     if ( ret instanceof $ ) {
190                         return null;
191                     }
192                     return ret;
193                 });
194             });
195 
196             return promise;
197         }

199         /** @private */
200         function page_open_callback_factory(done) {
201             return function() {
202                 diag('loaded: ' + page_eval(function() { return location.href; }));
203                 page_eval(function() {
204                     $(window).click(function(e) {
205                         var link = $(e.target).filter('a');
206                         if ( link.length ) {
207                             location = link.prop('href');
208                         }
209                     });
210                 });
211                 done();
212             };
213         }

236         /**
237          * Navigates to the given path. The path is considered relative to the
238          * <tt>base_url</tt> you set when creating the test object.
239          *
240          * @param {String} path The path to navigate to.
241          */
242         this.open = function(path) {
243             queue_async('open', function(done) {
244                 if ( ! page ) {
245                     page = require('webpage').create();
246                     page.viewportSize = { width: opt.width, height: opt.height };
247                     page.onConsoleMessage = function(message) {
248                         diag('console> ' + message, 1);
249                     };
250                 }
251 
252                 // Override the page console.log object so we capture all params
253                 page.onInitialized = proxy(function() {
254                    page_eval(function() {
255                        (function(old) {
256                            console.log = function() {
257                                old.apply(this, [Array.prototype.slice.call(arguments).join(' ')]);
258                            };
259                        })(console.log);
260                    });
261                 });
262                 page.onLoadFinished = page_open_callback_factory(done);
263                 page.open(opt.base_url + path);
264             });
265         };

280         /**
281          * Tell the testlib that you are done with your tests.
282          *
283          * This will output a TAP summary of the tests run, and exit. Calling
284          * this lets TAP parsers know that your script finished (as opposed to
285          * crashing), so make sure you call it.
286          */
287         this.done = function() {
288             queue_async('done', function(done) {
289                 var exit_code = 0;
290                 if ( tests.total ) {
291                     console.log('1..' + tests.total);
292                     if ( tests.failed ) {
293                         diag('Looks like you failed ' + tests.failed + ' test(s) of ' + tests.total + '.');
294                         exit_code = 1;
295                     }
296                 }
297                 else {
298                     console.log('1..0');
299                     diag('No tests run!');
300                 }
301 
302                 page.release();
303                 phantom.exit(exit_code);
304             });
305         };

314         /**
315          * Check that the one value matches another.
316          *
317          * Typically, you would call this method like so:
318          *
319          * <pre>t.is(t.text('title'), 'My Title', 'Page title is correct');</pre>
320          *
321          * @param got The value to check - typically you should use the result
322          *            of a call to {@link #text}, {@link #val}, {@link #attr}
323          *            or similar here.
324          * @param expected The value you expect. Will be compared using ==.
325          * @param {String} description The test description.
326          */
327         this.is = function(got, expected, description) {
328             queue_sync('is', function() {
329                 got = page_eval(got);
330                 expected = page_eval(expected);
331                 ok(got == expected, description, got, expected);
332             });
333         };
