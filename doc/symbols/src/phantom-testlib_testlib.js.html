  8     /**
  9      * Represents a value that will eventually be set.
 10      * @class
 11      * @private
 12      */
 13     var Promise = function() {
 14         var resolved = false, value;
 15         this.__defineGetter__('resolved', function() {
 16             return resolved;
 17         });
 18         this.__defineGetter__('value', function() {
 19             if ( ! resolved ) {
 20                 throw "Unresolved promise queried";
 21             }
 22             return value;
 23         });
 24         this.__defineSetter__('value', function(val) {
 25             resolved = true;
 26             value = val;
 27         });
 28     };

 30     /**
 31      * A library for writing web application tests in javascript.
 32      *
 33      * Provides a procedural interface for writing tests, so you don't have to
 34      * grok event based programming. Uses PhantomJS to provide a webkit-based
 35      * headless browser to run your tests against.
 36      *
 37      * Synopsis:
 38      * <pre>phantom.injectJs('testlib.js');
 39      * var t = new Test('https://github.com');
 40      * t.open('/');
 41      * t.is(t.text('title'), 'GitHub Â· Social Coding', 'Homepage loaded');</pre>
 42      * t.done();
 43      *
 44      * @param {String} base_url The base URL that all {@link #open} calls will be relative to.
 45      * @class
 46      */
 47     var Test = function(base_url) {
 48         var running = false;
 49         var page;
 50         var opt = {
 51             base_url: null,
 52             width: 1024,
 53             height: 768,
 54             timeout: 10000
 55         };
 56         var tests = {
 57             total: 0,
 58             ok: 0,
 59             failed: 0
 60         };
 61         var queue = [];
 62 
 63         // initialise
 64         opt.base_url = base_url;

 66         /** @private */
 67         function proxy(func) {
 68             var self = this;
 69             return function() { return func.apply(self, arguments); };
 70         }

 72         /** @private */
 73         function ok(success, description, got, expected) {
 74             var n = ++tests.total;
 75             if ( success ) {
 76                 console.log('ok ' + n + ' ' + description);
 77                 tests.ok++;
 78             }
 79             else {
 80                 console.log('not ok ' + n + ' ' + description);
 81                 if ( arguments.length == 4 ) {
 82                     diag("Got: " + got + "\nExpected: " + expected, 1);
 83                 }
 84                 tests.failed++;
 85             }
 86         }

 88         /** @private */
 89         function diag(message, indent) {
 90             indent = indent || 0;
 91             indent = Array(1+indent).join("\t");
 92             message.split(/\r?\n/).forEach(function(line) {
 93                 console.log('# ' + indent + line);
 94             });
 95         }

 97         /** @private */
 98         function job_done_factory(source) {
 99             var done = false;
100             return function() {
101                 if ( done ) {
102                     // We will not be done twice!
103                     return;
104                 }
105                 //console.log('job done: ' + source);
106                 running = false;
107                 done = true;
108                 setTimeout(run_queue, 0);
109             };
110         }
111 
112         /** @private */
113         function run_queue() {
114             if ( running || queue.length === 0 ) {
115                 // Nothing needs doing
116                 return;
117             }
118             running = true;
119             var job = queue.shift();
120             try {
121                 if ( job.is_async ) {
122                     var done = job_done_factory(job.func);
123                     var timeout = setTimeout(function() {
124                         ok(false, job.func + ' timed out');
125                         done();
126                     }, opt.timeout);
127                     job.func.call(this, function() {
128                         clearTimeout(timeout);
129                         done();
130                     });
131                 }
132                 else {
133                     job.func.call(this);
134                     setTimeout(job_done_factory(job.func), 0);
135                 }
136             }
137             catch(e) {
138                 console.log('Internal error: ' + e);
139                 phantom.exit(127);
140             }
141         }

143         /** @private */
144         function queue_async(name, func) {
145             /** @ignore */
146             func.toString = function() { return name; };
147             queue.push({
148                 is_async: true,
149                 func: func
150             });
151             run_queue();
152         }

154         /** @private */
155         function queue_sync(name, func) {
156             /** @ignore */
157             func.toString = function() { return name; };
158             queue.push({
159                 is_async: false,
160                 func: func
161             });
162             run_queue();
163         }

165         /** @private */
166         function page_eval(arg) {
167             if ( typeof(arg) === 'function' ) {
168                 return page.evaluate(arg);
169             }
170             if ( arg instanceof Promise ) {
171                 return arg.value;
172             }
173             if ( typeof(arg) === 'string' ) {
174                 return arg;
175             }
176             throw "Can't page_eval type: " + typeof(arg);
177         }

179         /** @private */
180         function page_set_argument(arg) {
181             page.evaluate('function() { __testlib_argument = ' + JSON.stringify(arg) + '; }');
182         }

184         /** @private */
185         function invoke_jquery() {
186             var promise = new Promise();
187             var args = Array.prototype.slice.call(arguments);
188             var method = args.shift();
189             var selector = args.shift();
190 
191             queue_sync('invoke_jquery', function() {
192                 page_set_argument({
193                     selector: selector,
194                     method: method,
195                     args: args
196                 });
197                 promise.value = page_eval(function() {
198                     var arg = __testlib_argument;
199                     var obj = $(arg.selector);
200                     var ret = obj[arg.method].apply(obj, arg.args);
201                     if ( ret instanceof $ ) {
202                         return null;
203                     }
204                     return ret;
205                 });
206             });
207 
208             return promise;
209         }

211         /** @private */
212         function page_open_callback_factory(done) {
213             return function() {
214                 diag('loaded: ' + page_eval(function() { return location.href; }));
215                 page_eval(function() {
216                     $(window).click(function(e) {
217                         var link = $(e.target).filter('a');
218                         if ( link.length ) {
219                             location = link.prop('href');
220                         }
221                     });
222                 });
223                 done();
224             };
225         }

261         /**
262          * Navigates to the given path. The path is considered relative to the
263          * <tt>base_url</tt> you set when creating the test object.
264          *
265          * @param {String} path The path to navigate to.
266          */
267         this.open = function(path) {
268             queue_async('open', function(done) {
269                 if ( ! page ) {
270                     page = require('webpage').create();
271                     page.viewportSize = { width: opt.width, height: opt.height };
272                     /** @ignore */
273                     page.onConsoleMessage = function(message) {
274                         diag('console> ' + message, 1);
275                     };
276                 }
277 
278                 // Override the page console.log object so we capture all params
279                 page.onInitialized = proxy(function() {
280                    page_eval(function() {
281                        (function(old) {
282                            /** @ignore */
283                            console.log = function() {
284                                old.apply(this, [Array.prototype.slice.call(arguments).join(' ')]);
285                            };
286                        })(console.log);
287                    });
288                 });
289                 page.onLoadFinished = page_open_callback_factory(done);
290                 page.open(opt.base_url + path);
291             });
292         };

294         /**
295          * Pauses the test execution.
296          *
297          * Note that this has many legitimate uses, but you shouldn't use it if
298          * there is a 'wait' method available that can do the job for you, as
299          * they return as soon as the event they're waiting for has happened.
300          *
301          * <pre>t.sleep(1000); // sleep for a second</pre>
302          *
303          * @param {Integer} ms How many milliseconds to sleep for.
304          */
305         this.sleep = function(ms) {
306             queue_async('sleep', function(done) {
307                 console.log('sleeping for ' + ms + 'ms');
308                 setTimeout(function() { console.log('finished sleeping'); done(); }, ms);
309             });

328         /**
329          * Tell the testlib that you are done with your tests.
330          *
331          * This will output a TAP summary of the tests run, and exit. Calling
332          * this lets TAP parsers know that your script finished (as opposed to
333          * crashing), so make sure you call it.
334          */
335         this.done = function() {
336             queue_async('done', function(done) {
337                 var exit_code = 0;
338                 if ( tests.total ) {
339                     console.log('1..' + tests.total);
340                     if ( tests.failed ) {
341                         diag('Looks like you failed ' + tests.failed + ' test(s) of ' + tests.total + '.');
342                         exit_code = 1;
343                     }
344                 }
345                 else {
346                     console.log('1..0');
347                     diag('No tests run!');
348                 }
349 
350                 page.release();
351                 phantom.exit(exit_code);
352             });
353         };

355         /**
356          * Check that the one value matches a regular expression.
357          *
358          * Typically, you would call this method like so:
359          *
360          * <pre>t.like(t.text('title'), /Title/, 'Page title contains "Title"');</pre>
361          *
362          * @param got The value to check - typically you should use the result
363          *            of a call to {@link #text}, {@link #val}, {@link #attr}
364          *            or similar here.
365          * @param expected The regex this value should match.
366          * @param {String} description The test description.
367          */
368         this.like = function(got, expected, description) {
369             queue_sync('like', function() {
370                 got = page_eval(got);
371                 ok(expected.test(got), description, got, expected);
372             });
373         };

375         /**
376          * Check that the one value matches another.
377          *
378          * Typically, you would call this method like so:
379          *
380          * <pre>t.is(t.text('title'), 'My Title', 'Page title is correct');</pre>
381          *
382          * @param got The value to check - typically you should use the result
383          *            of a call to {@link #text}, {@link #val}, {@link #attr}
384          *            or similar here.
385          * @param expected The value you expect. Will be compared using ==.
386          * @param {String} description The test description.
387          */
388         this.is = function(got, expected, description) {
389             queue_sync('is', function() {
390                 got = page_eval(got);
391                 expected = page_eval(expected);
392                 ok(got == expected, description, got, expected);
393             });
394         };
